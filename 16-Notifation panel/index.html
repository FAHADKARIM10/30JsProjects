How It Works

The complete, production-ready toast engine powering this demo.

// ── ToastKit Core Engine ────────────────────────── let toastBox = document.getElementById("toastBox"); let duration = 3500; // auto-dismiss time in ms let position = 'top-right'; let activeToasts = 0; const CONFIG = { success: { icon: '✅', title: 'Success' }, error: { icon: '❌', title: 'Error' }, warning: { icon: '⚠️', title: 'Warning' }, info: { icon: 'ℹ️', title: 'Info' }, }; function showToast(msg, type = 'info') { const cfg = CONFIG[type]; const toast = document.createElement('div'); toast.classList.add('toast', type); toast.innerHTML = ` <div class="toast-icon">${cfg.icon}</div> <div class="toast-body"> <div class="toast-title">${cfg.title}</div> <div class="toast-msg">${msg}</div> </div> <div class="toast-close">✕</div> <div class="toast-progress"></div> `; // Click to dismiss toast.querySelector('.toast-close') .addEventListener('click', () => removeToast(toast)); toastBox.appendChild(toast); // Animate progress bar const bar = toast.querySelector('.toast-progress'); bar.style.width = '100%'; setTimeout(() => bar.style.width = '0%', 50); bar.style.transition = `width ${duration}ms linear`; // Auto-dismiss setTimeout(() => removeToast(toast), duration); } function removeToast(toast) { toast.classList.add('removing'); setTimeout(() => toast.remove(), 350); }
